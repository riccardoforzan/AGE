SIZE_LIMIT = 100 * 1024 * 1024  # 100 MB


def is_file_too_large(filepath: str) -> bool:
    size = os.path.getsize(str(filepath))
    return int(size) >= int(SIZE_LIMIT)


def is_html(filepath: str) -> bool:
    "<!doctype html"

    mt = magic.from_file(filepath).lower()
    if "html" in mt:
        return True

    with open(filepath, "r") as f:
        try:
            return "<!doctype html" in f.read().lower()
        except Exception:
            return False


def is_json(filepath: str) -> bool:
    with open(filepath, "r") as f:
        try:
            json.load(f)
            return True
        except Exception:
            return False

import time
from pathlib import Path


def delete_file(file_path: str):
    if os.path.isfile(file_path):
        print(f"Deleting {file_path}")
        os.remove(file_path)


def change_extension(filepath: str, new_extension: str):
    suffix = Path(filepath).suffix

    new_name = None

    if len(suffix) > 0:
        new_name = filepath.replace(suffix, f".{new_extension}")
    else:
        new_name = f"{filepath}.{new_extension}"

    if os.path.isfile(new_name):
        time_str = time.strftime("%Y_%m_%d-%I_%M_%S")
        new_name = f"{time_str}-{new_name}"

    print(f"Renaming {filepath} to {new_name}")
    os.rename(filepath, new_name)
